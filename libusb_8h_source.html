156 static inline uint16_t libusb_cpu_to_le16(const uint16_t x)
157 {
158  union {
159  uint8_t b8[2];
160  uint16_t b16;
161  } _tmp;
162  _tmp.b8[1] = (uint8_t) (x >> 8);
163  _tmp.b8[0] = (uint8_t) (x & 0xff);
164  return _tmp.b16;
165 }
